async function atualizarGPS() { try { const res = await fetch(url); if (!res.ok) { throw new Error("HTTP error " + res.status); } const data = await res.json(); if (!data) { document.getElementById("status").innerText = "⚠️ Firebase vazio"; return; } const chaves = Object.keys(data); const ultimaChave = chaves[chaves.length - 1]; const ultimoDado = data[ultimaChave]; const lat = parseFloat(ultimoDado.lat); const lng = parseFloat(ultimoDado.lng); const pos = [lat, lng]; if (!marker) { marker = L.marker(pos).addTo(map); } else { marker.setLatLng(pos); } map.setView(pos, 18); document.getElementById("status").innerText = `📍 ${lat} | ${lng}`; } catch (err) { console.error(err); document.getElementById("status").innerText = "❌ Erro ao buscar Firebase (ver console)"; } }
